#Bin dereplication using DAS_Tool
module purge
module load DAS_Tool/1.1.5-gimkl-2022a-R-4.2.1

# Create DAS_Tool output directory
mkdir -p DAS_Tool/

# Run DAS_Tool
DAS_Tool -i metabat_associations.txt,maxbin_associations.txt -l MetaBAT,MaxBin -t 10 --write_bins --search_engine diamond -c /scale_wlg_nobackup/filesets/nobackup/lincoln03750/MGS_RDF/CO_ASSEMBLY/Assembly/Glenfern/final.contigs_m1500.fa -o DAS_Tool/

#Run CheckM2 to assess bin completeness and contamination
module purge
module load CheckM2/1.0.1-Miniconda3

checkm2 predict -t 10 -x .fa --input DAS_Tool/_DASTool_bins/ --output-directory DAS_Tool/CheckM2 --force
