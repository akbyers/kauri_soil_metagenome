# Navigate to working directory
cd /scale_wlg_nobackup/filesets/nobackup/lincoln03750/MGS_RDF/CO_ASSEMBLY/
#mkdir Binning/Glenfern

# Load seqmagick
#module purge
#module load seqmagick/0.8.4-gimkl-2020a-Python-3.8.2

# Filter assemblies
#seqmagick convert --min-length 1500 Assembly/Glenfern/final.contigs.fa Assembly/Glenfern/final.contigs_m1500.fa

#Obtain coverage profiles via read mapping
module purge
module load Bowtie2/2.5.4-GCC-12.3.0

# Build index
bowtie2-build Assembly/Glenfern/final.contigs_m1500.fa Assembly/Glenfern/bw_spades

# Map the reads
module purge
module load Bowtie2/2.5.4-GCC-12.3.0 SAMtools/1.19-GCC-12.3.0

bowtie2 --minins 200 --maxins 800 --threads 20 --sensitive -x Assembly/Glenfern/bw_spades -1 /scale_wlg_nobackup/filesets/nobackup/lincoln03750/MGS_RDF/raw_data_Aotea/concatenated_reads/GF_CONCAT_1.fastq -2 /scale_wlg_nobackup/filesets/nobackup/lincoln03750/MGS_RDF/raw_data_Aotea/concatenated_reads/GF_CONCAT_2.fastq -S Binning/Glenfern/Glenfern.sam

samtools sort --threads 12 -o Binning/Glenfern/Glenfern.bam Binning/Glenfern/Glenfern.sam

#Binning using METABAT2
module purge
module load MetaBAT/2.15-GCC-11.3.0

# Manual specification of files
jgi_summarize_bam_contig_depths --outputDepth Binning/Glenfern/metabat.txt Binning/Glenfern/Glenfern.bam
metabat2 -t 12 -m 1500 -i Assembly/Glenfern/final.contigs_m1500.fa -a Binning/Glenfern/metabat.txt -o Binning/Glenfern/metabat/metabat

#Binning using MaxBin
module purge
module load MaxBin/2.2.7-GCC-11.3.0-Perl-5.34.1

#format MaxBin coverage file
cut -f1,4,6,8,10 Binning/Glenfern/metabat.txt > Binning/Glenfern/maxbin.txt

# Output directory
mkdir -p Binning/Glenfern/maxbin/

# Run MaxBin
run_MaxBin.pl -thread 12 -min_contig_length 1500 -contig Assembly/Glenfern/final.contigs_m1500.fa -abund Binning/Glenfern/maxbin.txt -out Binning/Glenfern/maxbin/maxbin -max_iteration 30
