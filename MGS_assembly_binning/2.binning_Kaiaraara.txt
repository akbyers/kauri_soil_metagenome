cd /scale_wlg_nobackup/filesets/nobackup/lincoln03750/MGS_RDF/
#mkdir Binning/Kaiaraara

# Load seqmagick
#module purge
#module load seqmagick/0.8.4-gimkl-2020a-Python-3.8.2

# Filter assemblies
#seqmagick convert --min-length 1500 Assembly/Kaiaraara/final.contigs.fa Assembly/Kaiaraara/final.contigs_m1500.fa

#Obtain coverage profiles via read mapping
#module purge
#module load Bowtie2/2.5.4-GCC-12.3.0

# Build index
#bowtie2-build CO_ASSEMBLY/Assembly/Kaiaraara/final.contigs_m1500.fa CO_ASSEMBLY/Assembly/Kaiaraara/bw_spades

# Map the reads
module purge
module load Bowtie2/2.5.4-GCC-12.3.0 SAMtools/1.19-GCC-12.3.0

bowtie2 --minins 200 --maxins 800 --threads 20 --sensitive -x CO_ASSEMBLY/Assembly/Kaiaraara/bw_spades -1 raw_data_ALL/concatenated_reads/KA_CONCAT_1.fq -2 raw_data_ALL/concatenated_reads/KA_CONCAT_2.fq -S CO_ASSEMBLY/Binning/Kaiaraara/Kaiaraara.sam

samtools sort --threads 20 -o CO_ASSEMBLY/Binning/Kaiaraara/Kaiaraara.bam CO_ASSEMBLY/Binning/Kaiaraara/Kaiaraara.sam

#Binning using METABAT2
module purge
module load MetaBAT/2.15-GCC-11.3.0

cd /scale_wlg_nobackup/filesets/nobackup/lincoln03750/MGS_RDF/CO_ASSEMBLY/

# Manual specification of files
jgi_summarize_bam_contig_depths --outputDepth Binning/Kaiaraara/metabat.txt Binning/Kaiaraara/Kaiaraara.bam
metabat2 -t 20 -m 1500 -i Assembly/Kaiaraara/final.contigs_m1500.fa -a Binning/Kaiaraara/metabat.txt -o Binning/Kaiaraara/metabat/metabat

#Binning using MaxBin
module purge
module load MaxBin/2.2.7-GCC-11.3.0-Perl-5.34.1

#format MaxBin coverage file
cut -f1,4,6,8,10 Binning/Kaiaraara/metabat.txt > Binning/Kaiaraara/maxbin.txt

# Output directory
mkdir -p Binning/Kaiaraara/maxbin/

# Run MaxBin
run_MaxBin.pl -thread 20 -min_contig_length 1500 -contig Assembly/Kaiaraara/final.contigs_m1500.fa -abund Binning/Kaiaraara/maxbin.txt -out Binning/Kaiaraara/maxbin/maxbin -max_iteration 30
